## Domain Mapping
- "KB_IT_CYBER"   → PDPA, Cybersecurity, Data Governance, IT Policy
- "KB_HR_PROC"    → บุคลากร, จัดซื้อจัดจ้าง, งบประมาณ, สัญญา
- "KB_OPERATIONS" → ความปลอดภัย, มาตรฐานช่าง, กฎกระทรวง
- "KB_COMPLIANCE" → ISO 37301, CMS, Risk Management, Compliance Culture

## Query Type (สำคัญสำหรับ Hybrid Structure)
- "hierarchy"  → ถามเกี่ยวกับ "มีเอกสารอะไรบ้าง", "ชุดนี้ประกอบด้วย"
                 → Merger จะเดิน children tree รวบรวม doc_links
- "xref"       → ถามเกี่ยวกับ "ข้อนี้ทำยังไง", "ขั้นตอน", "วิธีปฏิบัติ"
                 → Merger จะ trigger Secondary KB ตาม XREF comment
- "combined"   → ถามแบบรวม เช่น "สรุปสิ่งที่ต้องทำ"
                 → ใช้ทั้งสองชั้น
- "direct"     → ถามตรง ไม่ต้องใช้ Hierarchy หรือ XREF พิเศษ

## Cross-KB Auto-Detection
เพิ่ม KB_COMPLIANCE ใน kb_targets ถ้า:
1. intent = "procedure" และ domain = KB_IT_CYBER
2. คำถามมีคำ "แนวปฏิบัติ" / "ขั้นตอน" / "ISO"

Variable Assigner — Python

python — Node 2: parse orchestrator JSON + assign query_type
import json

def main(orchestrator_json: str) -> dict:
    try:
        data = json.loads(orchestrator_json)
    except:
        data = {"domain":"KB_COMPLIANCE", "kb_targets":["KB_COMPLIANCE"],
                "intent":"general", "query_type":"direct", "needs_hyde":True}

    kb_targets = data.get("kb_targets", [data.get("domain", "KB_COMPLIANCE")])

    return {
        "domain":        data.get("domain", "KB_COMPLIANCE"),
        "intent":        data.get("intent", "general"),
        "query_type":    data.get("query_type", "direct"),
        "primary_kb":    kb_targets[0] if len(kb_targets) > 0 else "KB_COMPLIANCE",
        "secondary_kb":  kb_targets[1] if len(kb_targets) > 1 else "",
        "has_secondary": len(kb_targets) > 1,
        "needs_hyde":    data.get("needs_hyde", True),
        "xref_kb_detected": ""   # จะถูกอัปเดตโดย Merger หลัง Retrieval #1
    }

Context Merger + Hierarchy Resolver — Python

🔑 Hybrid Core — จุดสำคัญที่สุดของ Workflow
query_type ควบคุมทุกอย่าง: hierarchy → อ่าน children · xref → อ่าน cross_refs + ตรวจ KB ข้าม · combined → ทำทั้งสอง · direct → merge ปกติ
python — Node 7: Merger + Hierarchy Resolver
def main(
    primary_chunks:   list,
    secondary_chunks: list,
    primary_kb:       str,
    secondary_kb:     str,
    query_type:       str    # "hierarchy" | "xref" | "combined" | "direct"
) -> dict:

    all_chunks   = []
    seen         = set()
    xref_kb_set  = set()   # KB ที่ต้องดึงเพิ่มจาก XREF comment

    # ── รวม Chunks ───────────────────────────────────────────
    for chunk in (primary_chunks or []):
        h = chunk.get("content", "")[:100]
        if h not in seen:
            chunk["kb_source"] = primary_kb
            chunk["weight"]    = 1.0
            all_chunks.append(chunk); seen.add(h)

    for chunk in (secondary_chunks or []):
        h = chunk.get("content", "")[:100]
        if h not in seen:
            chunk["kb_source"] = secondary_kb
            chunk["weight"]    = 0.85
            all_chunks.append(chunk); seen.add(h)

    all_chunks.sort(
        key=lambda c: c.get("score", 0) * c.get("weight", 1.0),
        reverse=True
    )
    top = all_chunks[:7]

    # ── Build Context + Resolve Hybrid Metadata ──────────────
    formatted     = []
    direct_links  = []
    hier_links    = []
    xref_links    = []
    seen_titles   = set()

    for i, chunk in enumerate(top):
        meta    = chunk.get("metadata", {})
        title   = meta.get("doc_title", "ไม่ระบุ")
        link    = meta.get("doc_link", "")
        kb_src  = chunk.get("kb_source", "")
        article = meta.get("article", "")
        content = chunk.get("content", "")

        ref = title + (f" · {article}" if article else "")
        formatted.append(f"[{i+1}] {ref} ({kb_src})\n{content}")

        # Direct links — เอกสารที่ถูก Retrieve โดยตรง
        if link and title not in seen_titles:
            direct_links.append({"title": title, "link": link, "kb": kb_src})
            seen_titles.add(title)

        # ── ชั้นที่ 1: Hierarchy Resolver ─────────────────────
        if query_type in ("hierarchy", "combined"):
            hier = meta.get("hierarchy", {})
            for child in hier.get("children", []):
                ct = child.get("doc_title", "")
                cl = child.get("doc_link", "")
                ck = child.get("kb", "")
                if cl and ct not in seen_titles:
                    hier_links.append({"title": ct, "link": cl, "kb": ck})
                    seen_titles.add(ct)

        # ── ชั้นที่ 2: Cross-Reference Resolver ──────────────
        if query_type in ("xref", "combined"):
            for xref in meta.get("cross_refs", []):
                xk = xref.get("kb", "")
                xl = xref.get("doc_link", "")
                xt = xref.get("doc_title", "")
                if xk and xk != primary_kb:
                    xref_kb_set.add(xk)   # แจ้ง KB ที่ต้อง trigger เพิ่ม
                if xl and xt not in seen_titles:
                    xref_links.append({"title": xt, "link": xl, "kb": xk})
                    seen_titles.add(xt)

    # ── Format Links ─────────────────────────────────────────
    def fmt(lst):
        return "\n".join([f"- 📄 [{d['title']}]({d['link']}) `{d['kb']}`" for d in lst])

    return {
        "merged_context":   "\n---\n".join(formatted),
        "all_doc_links":    fmt(direct_links + xref_links),
        "hierarchy_links":  fmt(hier_links) if hier_links else "",
        "xref_kb_detected": ",".join(xref_kb_set),
        "kb_used":          ", ".join({c["kb_source"] for c in top})
    }

Citation Formatter — Python

python — Node 9: แปลง citation เป็น Markdown links
import re

def main(
    answer_raw:       str,
    primary_chunks:   list,
    secondary_chunks: list
) -> dict:

    # สร้าง lookup: doc_title → doc_link
    link_map = {}
    for chunk in (primary_chunks or []) + (secondary_chunks or []):
        meta  = chunk.get("metadata", {})
        title = meta.get("doc_title", "")
        link  = meta.get("doc_link", "")
        if title and link:
            link_map[title] = link
            # เพิ่ม children links เข้า map ด้วย
            hier = meta.get("hierarchy", {})
            for child in hier.get("children", []):
                ct = child.get("doc_title", "")
                cl = child.get("doc_link", "")
                if ct and cl:
                    link_map[ct] = cl

    # แทน [ชื่อเอกสาร · ข้อ X] ด้วย Markdown link
    def replace_ref(match):
        inner    = match.group(1)
        doc_name = inner.split(" · ")[0].strip()
        if doc_name in link_map:
            return f"[{inner}]({link_map[doc_name]})"
        return match.group(0)

    pattern = r'\[([^\]]+·[^\]]+)\](?!\()'
    final   = re.sub(pattern, replace_ref, answer_raw)

    return {"final_answer": final, "link_count": len(link_map)}
