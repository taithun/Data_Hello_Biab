import json

def main(orchestrator_json: str) -> dict:
    try:
        data = json.loads(orchestrator_json)
    except:
        # Fallback ถ้า LLM ส่ง JSON ไม่ครบ
        data = {
            "domain": "KB_COMPLIANCE",
            "kb_targets": ["KB_COMPLIANCE"],
            "intent": "general",
            "needs_hyde": True
        }

    kb_targets = data.get("kb_targets", [data.get("domain", "KB_COMPLIANCE")])

    return {
        "domain":        data.get("domain", "KB_COMPLIANCE"),
        "intent":        data.get("intent", "general"),
        "primary_kb":    kb_targets[0] if len(kb_targets) > 0 else "KB_COMPLIANCE",
        "secondary_kb":  kb_targets[1] if len(kb_targets) > 1 else "",
        "tertiary_kb":   kb_targets[2] if len(kb_targets) > 2 else "",
        "has_secondary": len(kb_targets) > 1,
        "needs_hyde":    data.get("needs_hyde", True),
        "xref_reason":   data.get("xref_reason", "")
    }

4. คำถามถามทั้ง "หลักการ" และ "วิธีทำ" → ต้องดึงจาก 2 KB เสมอ

Context Merger — Python Code

python — Node 7: Context Merger
def main(
    primary_chunks: list,
    secondary_chunks: list,
    primary_kb: str,
    secondary_kb: str
) -> dict:

    all_chunks = []
    seen_content = set()

    # Primary KB — weight 1.0
    for chunk in (primary_chunks or []):
        content_hash = chunk.get("content", "")[:100]
        if content_hash not in seen_content:
            chunk["kb_source"] = primary_kb
            chunk["weight"] = 1.0
            all_chunks.append(chunk)
            seen_content.add(content_hash)

    # Secondary KB — weight 0.85
    for chunk in (secondary_chunks or []):
        content_hash = chunk.get("content", "")[:100]
        if content_hash not in seen_content:
            chunk["kb_source"] = secondary_kb
            chunk["weight"] = 0.85
            all_chunks.append(chunk)
            seen_content.add(content_hash)

    # เรียงตาม score × weight, จำกัด 7 chunks
    all_chunks.sort(
        key=lambda c: c.get("score", 0) * c.get("weight", 1.0),
        reverse=True
    )
    top_chunks = all_chunks[:7]

    # Format Context + Doc Links
    formatted = []
    doc_links = []

    for i, chunk in enumerate(top_chunks):
        meta     = chunk.get("metadata", {})
        title    = meta.get("doc_title", "ไม่ระบุ")
        link     = meta.get("doc_link", "")
        kb_src   = chunk.get("kb_source", "")
        article  = meta.get("article", "")
        content  = chunk.get("content", "")

        ref = title
        if article:
            ref += f" · {article}"

        formatted.append(f"[{i+1}] {ref} ({kb_src})\n{content}")

        if link and title not in [d["title"] for d in doc_links]:
            doc_links.append({"title": title, "link": link, "kb": kb_src})

    links_md = "\n".join(
        [f"- 📄 [{d['title']}]({d['link']}) `{d['kb']}`" for d in doc_links]
    )
    kb_used_str = ", ".join(list({c["kb_source"] for c in top_chunks}))

    return {
        "merged_context":   "\n---\n".join(formatted),
        "doc_links_md":     links_md,
        "total_sources":    len(doc_links),
        "kb_used":          kb_used_str
    }

Citation Formatter — Python Code

python — Node 9: Citation Formatter
import re

def main(
    answer_raw: str,
    primary_chunks: list,
    secondary_chunks: list
) -> dict:

    # รวม chunks ทั้งหมดเพื่อสร้าง lookup table
    all_chunks = (primary_chunks or []) + (secondary_chunks or [])
    link_map = {}

    for chunk in all_chunks:
        meta  = chunk.get("metadata", {})
        title = meta.get("doc_title", "")
        link  = meta.get("doc_link", "")
        if title and link:
            link_map[title] = link

    # แทน [ชื่อเอกสาร · ข้อ X] ด้วย Markdown link ถ้ามี doc_link
    def replace_ref(match):
        full_ref = match.group(0)   # [ชื่อ · ข้อ X]
        inner    = match.group(1)   # ชื่อ · ข้อ X
        parts    = inner.split(" · ")
        doc_name = parts[0].strip()

        if doc_name in link_map:
            return f"[{inner}]({link_map[doc_name]})"
        return full_ref

    # Pattern: [text · text]
    pattern = r'\[([^\]]+·[^\]]+)\](?!\()'
    final_answer = re.sub(pattern, replace_ref, answer_raw)

    return {
        "final_answer": final_answer,
        "link_count":   len(link_map)
    }
